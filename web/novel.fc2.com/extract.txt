### 一覧ページから本の情報とリンク抽出
### CONTENT_ から始まる物は各話HTMLからの抽出用
###
### 書式: タブ区切り
###       (cssQueryはカンマで複数指定 前を優先 :の後にElemtnsの位置0～)
###内部ID	cssQuery1
###内部ID	cssQuery1,cssQuery2
###内部ID	cssQuery1:位置[,cssQuery2]
### または
###内部ID_REGEX	正規表現
###

################################
## FC2小説
## http://novel.fc2.com/

## 読み込み対象
PAGE_REGEX	^\/novel\.php\?mode=tc

################################
##一覧ページの抽出設定
## シリーズタイトル
#SERIES	
## タイトル
TITLE	.sh_heading_main_a:0
## 著者
AUTHOR	.username a:0,.username:0
## 説明
DESCRIPTION	.novel_comment:0

##表紙画像
COVER_IMG	.novel_img img:0
##COVER_HREF	.novel_img a:0


##各話へのリンクと更新日時取得用の外側のノード
##CHILD_NODE	

## 各話へのリンク cssQuery
HREF	.novel_subtitle a
## 各話へのリンク パターン
#HREF_REGEX	\/n\d+\w*\/\d+\/

##更新日時 比較文字列 (hrefに1:1で対応するタグ 内部のinnerHTML文字列で比較)
UPDATE	.noveldescription table td p:0


################################
###各話の抽出設定

## 各話 Chapter 先頭の章のみ大見出しで出力
#CONTENT_CHAPTER	.chapter_title
## 各話 章タイトル 中見出しで出力
CONTENT_SUBTITLE	.novel_subtitle
## 各話 本文
CONTENT_ARTICLE	.novel_body

##本文タグ外に画像がある場合本文の前に出力
CONTENT_IMG	.novel_img img

##前書き
#CONTENT_PREAMBLE	.novel_p .novel_view
## 補足
#CONTENT_APPENDIX	.novel_a .novel_view
